    table-align-text-post = false
}

%%%%%%%%%%%%%%%%%%%%%%%%
% PGFPlots
%%%%%%%%%%%%%%%%%%%%%%%%

\RequirePackage{pgfplots}
\usepgfplotslibrary{groupplots}
\usepgfplotslibrary{dateplot}
\usepgfplotslibrary{units}
\usepgfplotslibrary{colorbrewer}

% Necessary to use SI units in charts
% y unit = \meter
\pgfplotsset{
  unit code/.code 2 args=
    \begingroup
    \protected@edef\x{\endgroup\si{#1#2}}\x
}

% Set color index
\pgfplotsset{
    cycle list/Set1-9,
    cycle multiindex* list = {
        mark list*\nextlist
        linestyles*\nextlist
        Set1-9\nextlist
    },
}

% Set color index for bar plots
\pgfplotsset{
    /pgfplots/bar cycle list/.style={
        /pgfplots/cycle list={%
            {draw=Set1-A,fill=Set1-A!30!white,mark=none},%
            {draw=Set1-B,fill=Set1-B!30!white,mark=none},%
            {draw=Set1-C,fill=Set1-C!30!white,mark=none},%
            {draw=Set1-D,fill=Set1-D!30!white,mark=none},%
            {draw=Set1-E,fill=Set1-E!30!white,mark=none},%
            {draw=Set1-F!70!black,fill=Set1-F!30!white,mark=none},%
            {draw=Set1-G,fill=Set1-G!30!white,mark=none},%
            {draw=Set1-H,fill=Set1-H!30!white,mark=none},%
        }
    },
}

% Align left legend entries by default
\pgfplotsset{
    legend cell align = left,
}
